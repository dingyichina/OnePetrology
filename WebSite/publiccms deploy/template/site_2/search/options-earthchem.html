FROM item it join vmatreported mat on it.material_pkey = mat.material_pkey
WHERE
it.itemname!='age' and it.itemname!='age_min' and it.itemname!='age_max'

AND sourceint=8 and ST_Contains(st_geomfromtext('MULTIPOLYGON(((-37.9687499998311 45.4612438426453,-43.7695312498053 20.6693625972006,17.0507812499237 13.9194687762769,-37.9687499998311 45.4612438426453)))'),the_geom)
